CREATE TABLE IF NOT EXISTS Administrator(AdminID INTEGER, Fname VARCHAR(50), Lname VARCHAR(50), Username VARCHAR(50), Password VARCHAR(50));
CREATE TABLE IF NOT EXISTS Customer(CustomerID INTEGER, Fname VARCHAR(50), Lname VARCHAR(50), Username VARCHAR(50), Password VARCHAR(50), Address VARCHAR(50), City VARCHAR(50), State VARCHAR(50), ZipCode INTEGER, Email VARCHAR(50), PhoneNumber VARCHAR(50), CONSTRAINT Customer PRIMARY KEY (CustomerID));
CREATE TABLE IF NOT EXISTS FoodItem(FoodID INTEGER, Name VARCHAR(50), UnitPrice DECIMAL, CONSTRAINT FoodItem PRIMARY KEY(FoodID));
CREATE TABLE IF NOT EXISTS Cart(OrderID INTEGER, FoodID INTEGER, Quantity INTEGER, CONSTRAINT Cart FOREIGN KEY (FoodID) REFERENCES FoodItem(FoodID), CONSTRAINT Cart PRIMARY KEY(OrderID));
CREATE TABLE IF NOT EXISTS OrderHistory(CustomerID INTEGER, OrderID INTEGER, FoodID INTEGER, Quantity INTEGER, Ordered DATE, Total DECIMAL, CONSTRAINT OrderHistory FOREIGN KEY (OrderID) REFERENCES Cart, CONSTRAINT OrderHistory FOREIGN KEY (FoodID) REFERENCES FoodItem, CONSTRAINT OrderHistory FOREIGN KEY (CustomerID) REFERENCES Customer);
